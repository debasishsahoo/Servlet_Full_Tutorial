package jakarta.servlet;

import java.io.IOException;

/**
 * The Servlet interface defines methods that all servlets must implement.
 *
 * A servlet is a small Java program that runs within a Web server.
 * Servlets receive and respond to requests from Web clients, usually across HTTP.
 */
public interface Servlet {

    /**
     * Called by servlet container to initialize the servlet
     * @param config ServletConfig object containing servlet's configuration
     * @throws ServletException if initialization fails
     */
    void init(ServletConfig config) throws ServletException;

    /**
     * Called by servlet container to handle each request
     * @param req ServletRequest object containing client request
     * @param res ServletResponse object for sending response
     * @throws ServletException if servlet encounters difficulty
     * @throws IOException if I/O error occurs
     */
    void service(ServletRequest req, ServletResponse res)
        throws ServletException, IOException;

    /**
     * Called by servlet container when servlet is being destroyed
     */
    void destroy();

    /**
     * Returns ServletConfig object containing servlet's configuration
     * @return ServletConfig object
     */
    ServletConfig getServletConfig();

    /**
     * Returns information about servlet (author, version, copyright)
     * @return String containing servlet information
     */
    String getServletInfo();
}
